347     public static FontAwesomeBrands findByDescription(String description) {
348         for (FontAwesomeBrands font : values()) {
349             if (font.getDescription().equals(description)) {
350                 return font;
351             }
352         }
353         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
354     }
355 
356     private String description;
357     private char code;
358 
359     FontAwesomeBrands(String description, char code) {
360         this.description = description;
361         this.code = code;
362     }
363 
364     @Override
365     public String getDescription() {
366         return description;
367     }
368 
369     @Override
370     public char getCode() {
371         return code;
372     }
