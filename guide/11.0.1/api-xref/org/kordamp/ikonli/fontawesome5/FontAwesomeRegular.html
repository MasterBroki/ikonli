142     public static FontAwesomeRegular findByDescription(String description) {
143         for (FontAwesomeRegular font : values()) {
144             if (font.getDescription().equals(description)) {
145                 return font;
146             }
147         }
148         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
149     }
150 
151     private String description;
152     private char code;
153 
154     FontAwesomeRegular(String description, char code) {
155         this.description = description;
156         this.code = code;
157     }
158 
159     @Override
160     public String getDescription() {
161         return description;
162     }
163 
164     @Override
165     public char getCode() {
166         return code;
167     }
