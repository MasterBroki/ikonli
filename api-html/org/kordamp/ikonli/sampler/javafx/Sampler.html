129     private static class DemoTab extends Tab {
130         private DemoTab(Class<? extends Ikon> iconFontClass, EnumSet<? extends Ikon> enumSet) throws Exception {
131             super(iconFontClass.getSimpleName());
132             setClosable(false);
133 
134             GridPane pane = new GridPane();
135             ScrollPane scrollPane = new ScrollPane(pane);
136             setContent(scrollPane);
137 
138             int column = 0;
139             int row = 0;
140             int index = 0;
141             for (Ikon value : enumSet) {
142                 FontIcon icon = FontIcon.of(value);
143                 icon.getStyleClass().setAll("font-icon");
144                 pane.add(icon, column++, row);
145                 GridPane.setMargin(icon, new Insets(10, 10, 10, 10));
146                 if (++index % 10 == 0) {
147                     column = 0;
148                     row++;
149                 }
150             }
151         }
152     }
